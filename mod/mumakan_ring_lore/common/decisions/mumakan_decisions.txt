######################################################################
# Mumakan Ring-Lore & Shadow Sovereignty - Decisions
# Player-only (is_ai = no + ai_will_do base 0) to preserve balance and keep AI
# logic clean for multiplayer stability. Title loc key = decision id.
#
# Design note: requirements live in is_valid (not is_shown) so an eligible
# Mumakan/Black-Numenorean ruler ALWAYS sees the decision, greyed-out with the
# unmet requirement (independent / feudal-or-clan / Forges built) shown - clearer
# UX and avoids the decision silently disappearing.
######################################################################

##### PALIER 1 - Plonger dans le Savoir des Anneaux (forge the Minor Ring) #####
mumakan_decision_minor_ring = {
	ai_check_interval = 0
	decision_group_type = realms_in_exile_major
	title = mumakan_decision_minor_ring
	desc = mumakan_decision_minor_ring_desc
	selection_tooltip = mumakan_decision_minor_ring_tooltip
	confirm_text = mumakan_decision_minor_ring_confirm

	is_shown = {
		is_ai = no
		culture = {
			OR = {
				has_cultural_parameter = heritage_group_mumakani
				has_cultural_pillar = heritage_southern_adunaic
			}
		}
		NOT = { has_character_flag = has_forged_minor_ring }
	}

	is_valid = {
		# Independence + the Forges built. Government is already constrained to
		# non-tribal (feudal/clan) by the building's own construction requirement,
		# so no separate government check here (a bare OR block fails invisibly).
		is_independent_ruler = yes
		mumakan_has_forge_building_trigger = yes
	}

	cost = {
		piety = mumakan_minor_ring_piety_cost
		gold = mumakan_minor_ring_gold_cost
	}

	effect = {
		mumakan_forge_minor_ring_effect = yes
	}

	ai_potential = { always = no }
	ai_will_do = { base = 0 }
}
