######################## PALIER 1 - L'ANNEAU MINEUR ########################

mumakan_forge_minor_ring_effect = {
	add_character_flag = has_forged_minor_ring

	# Unlock the dynasty legacy track "La Volonte des Anneaux" globally for the house.
	dynasty ?= { set_variable = mumakan_anneaux_unlocked }

	# Create + equip the physical Minor Ring artifact.
	create_artifact = {
		name = mumakan_minor_ring_name
		description = mumakan_minor_ring_desc
		creator = root
		type = miscellaneous
		visuals = narya
		wealth = 100
		quality = 100
		template = mumakan_minor_ring_template
		modifier = mumakan_minor_ring_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact ?= {
		equip_artifact_to_owner_replace = yes
	}

	send_interface_toast = {
		title = mumakan_minor_ring_toast
		left_icon = root
	}
}

######################## PALIER 2 - L'ANNEAU MAITRE ########################

# Launch the forging: marks the safety flag and fires the RNG outcome event.
mumakan_forge_master_ring_effect = {

	add_character_flag = mumakan_master_ring_awakened
	# Robust bearer status + 16-ring counter (covers reaching this via console/RNG, not the decision).
	if = { limit = { NOT = { has_character_flag = mumakan_is_master_ring_bearer } } add_character_flag = mumakan_is_master_ring_bearer }
	if = { limit = { NOT = { has_variable = mumakan_loyalty_rings_forged } } set_variable = { name = mumakan_loyalty_rings_forged value = 0 } }

	create_artifact = {
		name = mumakan_master_ring_name
		description = mumakan_master_ring_desc
		creator = root
		type = miscellaneous
		visuals = vilya
		wealth = 100
		quality = 100
		template = mumakan_master_ring_template
		modifier = mumakan_master_ring_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact ?= {
		equip_artifact_to_owner_replace = yes
	}

	# Suprematie Absolue - Dread is pinned at 100 by the ring's dread_baseline_add;
	# this grants the full value immediately on awakening.
	add_dread = 100

	# Longevite Souveraine - immortality / stop aging.
	add_trait = mumakan_ringbound_immortal

	# Volonte de l'Akhash-Mumak is CONGENITAL - it runs through the whole bloodline.
	# Granted to the bearer + existing children; future descendants inherit it at
	# birth via the trait's genetic = yes (inherit_chance = 100).
	if = { limit = { NOT = { has_trait = will_of_the_ring } } add_trait = will_of_the_ring }
	every_child = {
		if = { limit = { NOT = { has_trait = will_of_the_ring } } add_trait = will_of_the_ring }
	}
}

######################## PALIER 3 - ANNEAUX DE LOYAUTE ########################

# Called from the loyalty-ring character interaction on_accept.
# scope:actor = the Master Ring bearer (liege), scope:recipient = the vassal.
mumakan_grant_loyalty_ring_effect = {
	scope:actor = {
		change_variable = { name = mumakan_loyalty_rings_forged add = 1 }

		# Begin the monthly tribute scheduler the first time a ring is gifted.
		if = {
			limit = { NOT = { has_character_flag = mumakan_tribute_scheduler_active } }
			add_character_flag = mumakan_tribute_scheduler_active
			trigger_event = { id = mumakan_ring.0020 days = 30 }
		}
	}

	scope:recipient = {
		# Serviteur de l'Anneau + immortality + fertility.
		if = { limit = { NOT = { has_trait = loyalty_to_the_ring } } add_trait = loyalty_to_the_ring }
		if = { limit = { NOT = { has_trait = mumakan_ringbound_immortal } } add_trait = mumakan_ringbound_immortal }
		add_character_modifier = mumakan_ringbound_servant_modifier
		add_character_flag = mumakan_ringbound_servant
		# Vanilla/RoE faction validity reads this flag: it makes the servant show up
		# in the game's "will not join a faction" list and blocks them joining OR
		# creating one (a true block, not just the yearly eject in the pulse).
		add_character_flag = joining_faction_block

		# Convert to the liege's customized Dark Cult and adopt the Pious archetype.
		set_character_faith = scope:actor.faith
		if = { limit = { NOT = { has_trait = zealous } } add_trait = zealous }

		# Locked +100 opinion toward the liege.
		add_opinion = {
			target = scope:actor
			modifier = mumakan_ring_of_loyalty_opinion
		}

		# Physical Ring of Loyalty artifact.
		create_artifact = {
			name = mumakan_loyalty_ring_name
			description = mumakan_loyalty_ring_desc
			creator = scope:actor
			type = miscellaneous
			visuals = ring_of_men
			wealth = 80
			quality = 80
			template = mumakan_loyalty_ring_template
			modifier = mumakan_loyalty_ring_modifier
			save_scope_as = newly_created_artifact
			decaying = no
		}
		scope:newly_created_artifact ?= {
			equip_artifact_to_owner_replace = yes
		}
	}
}

######################## SUCCESSION / INHERITANCE ########################

# Called from on_death (root = deceased). Transfers the Shadow Sovereignty chain
# to the primary heir so the bearer status survives succession.
mumakan_transfer_bearer_status_effect = {
	primary_heir ?= {
		add_character_flag = mumakan_is_master_ring_bearer
		save_scope_as = mumakan_new_bearer
		# The heir inherits the Master Ring, so they inherit its immortality too.
		if = {
			limit = { NOT = { has_trait = mumakan_ringbound_immortal } }
			add_trait = mumakan_ringbound_immortal
		}
		if = {
			limit = { NOT = { has_trait = will_of_the_ring } }
			add_trait = will_of_the_ring
		}
		# Carry over the per-bearer 16-ring counter (root = the deceased bearer in on_death).
		if = {
			limit = { root = { has_variable = mumakan_loyalty_rings_forged } }
			set_variable = { name = mumakan_loyalty_rings_forged value = root.var:mumakan_loyalty_rings_forged }
		}
		# Re-arm the tribute scheduler on the new bearer.
		if = {
			limit = { NOT = { has_character_flag = mumakan_tribute_scheduler_active } }
			add_character_flag = mumakan_tribute_scheduler_active
			trigger_event = { id = mumakan_ring.0020 days = 30 }
		}
	}
	# Re-bind the locked +100 opinion of every ring-bound servant to the NEW bearer.
	# (The old opinion targeted the now-dead bearer and goes inert; CK3 never retargets
	# opinions to heirs. root = deceased still holds these vassals at on_death.)
	if = {
		limit = { exists = scope:mumakan_new_bearer }
		every_vassal_or_below = {
			limit = {
				has_trait = loyalty_to_the_ring
				NOT = { has_opinion_modifier = { target = scope:mumakan_new_bearer modifier = mumakan_ring_of_loyalty_opinion } }
			}
			add_opinion = { target = scope:mumakan_new_bearer modifier = mumakan_ring_of_loyalty_opinion }
		}
	}
}

# Called from on_death of a ring-bound servant (root = deceased servant).
# loyalty_to_the_ring transmits to the primary heir at full weight.
mumakan_ringbound_servant_inherit_effect = {
	primary_heir ?= {
		if = { limit = { NOT = { has_trait = loyalty_to_the_ring } } add_trait = loyalty_to_the_ring }
		if = { limit = { NOT = { has_trait = mumakan_ringbound_immortal } } add_trait = mumakan_ringbound_immortal }
		add_character_modifier = mumakan_ringbound_servant_modifier
		add_character_flag = mumakan_ringbound_servant
		add_character_flag = joining_faction_block
	}
}

######################## MAINTENANCE PULSES ########################

# Faction immunity: a ring-bound servant cannot stand against the bearer.
# Two layers, no base-file override: (1) the joining_faction_block flag makes the
# game genuinely bar them from joining/creating factions and lists them under
# "will not join a faction"; (2) we still eject any faction they are already in.
# Called per-character from the yearly pulse (also retro-applies the flag to
# servants created before this update).
mumakan_faction_immunity_effect = {
	if = {
		limit = {
			has_trait = loyalty_to_the_ring
			NOT = { has_character_flag = joining_faction_block }
		}
		add_character_flag = joining_faction_block
	}
	if = {
		limit = {
			has_trait = loyalty_to_the_ring
			is_a_faction_member = yes
		}
		# 'leave_faction = yes' is invalid - the effect requires a FACTION target
		# (the log reported "expected faction"). Iterate every faction the character
		# has joined and leave each one explicitly. every_joined_faction yields
		# faction scope, so we save the character first and call leave_faction back
		# in character scope with the saved faction as the target.
		save_temporary_scope_as = mumakan_faction_leaver
		every_joined_faction = {
			save_temporary_scope_as = mumakan_target_faction
			scope:mumakan_faction_leaver = { leave_faction = scope:mumakan_target_faction }
		}
	}
}

# Regle d'Isolation Dynastique: if an immortal servant's house exceeds 15 living
# members, the servant FOUNDS A NEW CADET HOUSE under their own dynasty.
# create_cadet_branch is the scripted "Create Cadet Branch" interaction: it makes
# a new house WITHIN THE SAME DYNASTY with this character as its founder/head, and
# the descendants who follow them move into it. It does NOT strip claims, titles,
# or succession - it only prevents one house from bloating with immortal servants.
mumakan_cadet_scission_effect = {
	if = {
		limit = {
			has_trait = loyalty_to_the_ring
			# any_house_member is a dynasty_house-scoped iterator, so it must run
			# inside house = { ... } - calling it in character scope is the parse
			# error the log reported as create_cadet_branch "Expected opening bracket".
			house ?= { any_house_member = { count >= 16 is_alive = yes } }
			# is_dynasty_head is not a CK3 trigger; dynasty.dynast is the head.
			NOT = { this = dynasty.dynast }
		}
		create_cadet_branch = yes
	}
}
