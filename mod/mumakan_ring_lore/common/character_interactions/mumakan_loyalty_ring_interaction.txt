######################################################################
# Mumakan Ring-Lore & Shadow Sovereignty - Palier 3 interaction
# Offrir un Anneau de Loyaute. Right-click a powerful internal vassal (Duke+).
#
# is_shown is kept LIGHT (Mumakan/Black-Numenorean human ruler + a Duke+ internal
# vassal) so the interaction is ALWAYS visible in the Vassal menu; the real
# requirements (being a Master Ring bearer, affording the Piety, a valid target)
# live in is_valid_showing_failures_only so an unmet requirement shows greyed-out
# with its reason instead of the whole interaction silently disappearing.
######################################################################

mumakan_loyalty_ring_interaction = {
	icon = icon_gold
	category = interaction_category_vassal
	common_interaction = yes
	# A bare 'desc = <key>' is fine here: thousands of vanilla interactions use a bare
	# desc whose loc carries data functions. The desc is NOT the source of the
	# "Current level cannot be automatically flattened" error - that error comes from
	# the FAILURE-REASON tooltips generated by is_valid_showing_failures_only when a
	# condition is a complex scripted_trigger (nested OR/NOT) the engine cannot collapse
	# to a single line. The fix lives in is_valid_showing_failures_only below, where the
	# non-flattenable trigger is wrapped in custom_tooltip = { text = ... }.
	desc = mumakan_loyalty_ring_interaction_desc

	is_shown = {
		scope:actor = {
			is_ai = no
			culture = {
				OR = {
					has_cultural_parameter = heritage_group_mumakani
					has_cultural_pillar = heritage_southern_adunaic
				}
			}
		}
		scope:recipient != scope:actor
		scope:recipient = {
			is_landed = yes
			# character scope -> highest_held_title_tier, not the title-scope 'tier'.
			highest_held_title_tier >= tier_duchy
			is_vassal_or_below_of = scope:actor
		}
	}

	is_valid_showing_failures_only = {
		scope:actor = {
			# mumakan_can_forge_loyalty_ring_trigger has nested OR/NOT/has_variable logic
			# the failure-tooltip flattener cannot collapse into one line, which throws the
			# "Current level cannot be automatically flattened" error. custom_tooltip gives
			# it an explicit single-line failure reason (same idiom RoE uses, e.g.
			# funnel_resources_interaction: custom_tooltip = { text = ... monthly_character_income < 20 }).
			custom_tooltip = {
				text = mumakan_loyalty_ring_requires_master_ring_tt
				mumakan_can_forge_loyalty_ring_trigger = yes
			}
			piety >= mumakan_loyalty_ring_piety_cost
		}
		scope:recipient = {
			is_adult = yes
			is_alive = yes
			NOT = { has_trait = loyalty_to_the_ring }
		}
	}

	auto_accept = yes

	on_accept = {
		scope:actor = {
			add_piety = {
				value = mumakan_loyalty_ring_piety_cost
				multiply = -1
			}
		}
		mumakan_grant_loyalty_ring_effect = yes
	}

	ai_will_do = { base = 0 }
}
