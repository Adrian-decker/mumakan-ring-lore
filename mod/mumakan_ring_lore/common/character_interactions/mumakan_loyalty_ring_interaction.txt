######################################################################
# Mumakan Ring-Lore & Shadow Sovereignty - Palier 3 interaction
# Offrir un Anneau de Loyaute. Right-click a powerful internal vassal (Duke+).
#
# is_shown is kept LIGHT (Mumakan/Black-Numenorean human ruler + a Duke+ internal
# vassal) so the interaction is ALWAYS visible in the Vassal menu; the real
# requirements (being a Master Ring bearer, affording the Piety, a valid target)
# live in is_valid_showing_failures_only so an unmet requirement shows greyed-out
# with its reason instead of the whole interaction silently disappearing.
######################################################################

mumakan_loyalty_ring_interaction = {
	icon = icon_gold
	category = interaction_category_vassal
	common_interaction = yes
	desc = mumakan_loyalty_ring_interaction_desc

	is_shown = {
		scope:actor = {
			is_ai = no
			culture = {
				OR = {
					has_cultural_parameter = heritage_group_mumakani
					has_cultural_pillar = heritage_southern_adunaic
				}
			}
		}
		scope:recipient != scope:actor
		scope:recipient = {
			is_landed = yes
			# character scope -> highest_held_title_tier, not the title-scope 'tier'.
			highest_held_title_tier >= tier_duchy
			is_vassal_or_below_of = scope:actor
		}
	}

	is_valid_showing_failures_only = {
		scope:actor = {
			mumakan_can_forge_loyalty_ring_trigger = yes
			piety >= mumakan_loyalty_ring_piety_cost
		}
		scope:recipient = {
			is_adult = yes
			is_alive = yes
			NOT = { has_trait = loyalty_to_the_ring }
		}
	}

	auto_accept = yes

	on_accept = {
		scope:actor = {
			add_piety = {
				value = mumakan_loyalty_ring_piety_cost
				multiply = -1
			}
		}
		mumakan_grant_loyalty_ring_effect = yes
	}

	ai_will_do = { base = 0 }
}
